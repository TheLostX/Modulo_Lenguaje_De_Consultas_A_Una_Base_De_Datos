-- ejercicio practico
-- 1.- crear una base de datos
create database segundabd;

-- 2.- crear 2 tablas
create table persona(
	idPersona int auto_increment,
    nombre char,
    apellido varchar(30),
    rut varchar(20),
    primary key(idPersona)
);

create table mascota(
	idMascota int auto_increment,
    edad int,
    primary key (idMascota)
);
-- 3.-Agregar 2 nuevos campos a las tablas creadas
alter table persona add column edad int;
alter table mascota add column nombre varchar(30);

-- 4.- cambiar el tipo de dato a una columna
alter table persona modify nombre varchar(30);

-- 5.- Eliminar un campo de cada tabla creada
alter table persona drop column rut;
alter table mascota drop column edad;
